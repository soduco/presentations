\begin{frame}
  \frametitle{Traitement des données issues de l'analyse des annuaires}
  \textbf{AutoMatch} est un outil de mise en correspondance d'images
  \begin{itemize}
  \item[-] entre 1 image géoréférencée et un ensemble d'images à géoréférencer
  \item[-] identifie des points homologues (SIFT, SURF, ORB, AKAZE, BRISK)
  \item[-] calcule une transformation (homographie / RANSAC, polynomiale/hermert)
  \item[-] exporte les points et les images géoréférencées
  \end{itemize}
  \includegraphics[width=.6\textwidth]{automatch_poubelle_parcellaire}\\
  Un résultat entre l'Atlas du service du plan de 1888 et le Plan parcellaire municipal de Paris de la fin du XIXe
\end{frame}

\begin{frame}
  \frametitle{Traitement des données issues de l'analyse des annuaires}
  \textbf{AutoMatch} permet ou pourrait permettre de
  \begin{itemize}
  \item[-] assurer la traçabilité du processus de géoréférencement
  \item[-] re-géoréférencer des nouveaux scans
  \item[-] dé-géoréférencer des plans et reconstuire un atlas géoréférencé
  \item[-] mettre en correspondance des plans successifs (Atlas du service du plan)
  \item[-] géoréférencer à partir de sources multiples
  \end{itemize}
\end{frame}
